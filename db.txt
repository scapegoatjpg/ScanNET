
CREATE TABLE Customer (
  userID int NOT NULL,
  firstName varchar(255),
  lastName varchar(255),
  userName varchar(255),
  password varchar(255),
  IPAddress varchar(255),
  loginName varchar(255),
  invalidLogin int,
  bandwidth int,
  active varchar varchar(255),
  PRIMARY KEY (userID)
  FOREIGN KEY (IPAddress) REFERENCES Device (IPAddress)
);


CREATE TABLE NetworkActivity (
  activityReportID int NOT NULL,
  IPAddress varchar(255),
  networkNameSSID int NOT NULL,
  packetNum int NOT NULL,
  timeStamp time/date,
  source varchar(255),
  destination varchar(255),
  description varchar(255),
  maliciousDevices int ,
  actilvitySummary varchar(255),
  totalDevice int,
  PRIMARY KEY (activityReportID)
  FOREIGN KEY (IPAddress) REFERENCES Device (IPAddress)
  FOREIGN KEY (networkNameSSID) REFERENCES NetworkInfo (networkNameSSID)
);

 
CREATE TABLE NetworkInfo (
  networkNameSSID int NOT NULL,
  activityReportID int NOT NULL,
  IPAddress int NOT NULL,
  PHYType varchar(255),
  channelNum varchar(255),
  maxSpeed int,
  connectedDevices int,
  activeDevices int,
  PRIMARY KEY (networkNameSSID)
  FOREIGN KEY (activityReportID) REFERENCES NetworkActivity (activityReportID)
  FOREIGN KEY (IPAddress) REFERENCES Device (IPAddress)
);

CREATE TABLE Device (
  IPAddress varchar(255),
  MACAddress varchar(255),
  activityReportID int NOT NULL,
  deviceName varchar(255),
  customerName varchar(255),
  serviceProvider varchar(255),
  networkCard varchar(255),
  firstDetection date/time,
  lastDetection date/time,
  deviceActivity varchar(255),
  numOfMaliciousActivity int,
  PRIMARY KEY (IPAddress)
  FOREIGN KEY (activityReportID) REFERENCES NetworkActivity (activityReportID)
);
